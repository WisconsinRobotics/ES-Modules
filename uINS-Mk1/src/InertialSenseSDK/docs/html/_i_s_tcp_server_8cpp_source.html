46 int cISTcpServer::Open(const string& ipAddress, int port)
47 {
48  m_ipAddress = ipAddress;
49  m_port = port;
50  int status;
51  char portString[64];
52  SNPRINTF(portString, sizeof(portString), "%ld", (long)m_port);
53  addrinfo* result = NULL;
54  addrinfo hints = addrinfo();
55  hints.ai_family = AF_INET;
56  hints.ai_socktype = SOCK_STREAM;
57  hints.ai_protocol = IPPROTO_TCP;
58  hints.ai_flags = AI_PASSIVE;
59  status = getaddrinfo(m_ipAddress.length() == 0 ? NULL : m_ipAddress.c_str(), portString, &hints, &result);
60  if (status != 0)
61  {
62  Close();
63  return status;
64  }
65 
66  // setup socket
67  m_socket = socket(result->ai_family, result->ai_socktype, result->ai_protocol);
68  if (m_socket == 0)
69  {
70  freeaddrinfo(result);
71  Close();
72  return -1;
73  }
74 
75  // setup listener socket
76  status = ::bind(m_socket, result->ai_addr, (int)result->ai_addrlen);
77  if (status != 0)
78  {
79  freeaddrinfo(result);
80  Close();
81  return -1;
82  }
83 
84  freeaddrinfo(result);
85 
86  status = listen(m_socket, SOMAXCONN);
87  if (status != 0)
88  {
89  Close();
90  return -1;
91  }
92 
93  return status;
94 }

107 void cISTcpServer::Update()
108 {
109  uint8_t readBuff[8192];
110  int count;
111 
112  // connect up to 10 clients per loop
113  for (int i = 0; i < 10; i++)
114  {
115  struct timeval tv = { 0, 0 };
116  fd_set rs;
117  FD_ZERO(&rs);
118  FD_SET(m_socket, &rs);
119  int ret = select(0, &rs, NULL, NULL, &tv);
120  if (ret > 0)
121  {
122  socket_t socket = accept(m_socket, NULL, NULL);
123  if (socket != 0)
124  {
125  ISSocketSetBlocking(socket, false);
126  m_clients.push_back(socket);
127  }
128  }
129  else
130  {
131  break;
132  }
133  }
134 
135  for (size_t i = 0; i < m_clients.size(); i++)
136  {
137  if ((count = ISSocketRead(m_clients[i], readBuff, sizeof(readBuff))) < 0)
138  {
139  // remove the client
140  if (m_delegate != NULL)
141  {
142  m_delegate->OnClientDisconnected(m_clients[i]);
143  }
144  ISSocketClose(m_clients[i]);
145  m_clients.erase(m_clients.begin() + i--);
146  }
147  else if (count > 0)
148  {
149  if (m_delegate != NULL)
150  {
151  m_delegate->OnClientDataReceived(m_clients[i], readBuff, count);
152  }
153  }
154  }
155 }
