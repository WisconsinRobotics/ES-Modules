37 static void staticProcessRxData(CMHANDLE cmHandle, int pHandle, p_data_t* data)
38 {
39  (void)pHandle;
40  (void)cmHandle;
41 
42  if (data->hdr.id < DID_COUNT)
43  {
44  InertialSense::com_manager_cpp_state_t* s = (InertialSense::com_manager_cpp_state_t*)comManagerGetUserPointer(cmHandle);
45  pfnHandleBinaryData handler = s->binaryCallback[data->hdr.id];
46  s->stepLogFunction(s->inertialSenseInterface, data);
47  if (handler != NULL)
48  {
49  handler(s->inertialSenseInterface, data);
50  }
51 
52  pfnHandleBinaryData handlerGlobal = s->binaryCallbackGlobal;
53  if (handlerGlobal != NULL)
54  {
55  // Called for all DID's
56  handlerGlobal(s->inertialSenseInterface, data);
57  }
58 
59  // if we got the device info, populate it
60  if (data->hdr.id == DID_DEV_INFO)
61  {
62  memcpy(s->devInfo, data->buf, sizeof(dev_info_t));
63  }
64  else if (data->hdr.id == DID_CONFIG)
65  {
66  memcpy(s->config, data->buf, sizeof(config_t));
67  }
68  else if (data->hdr.id == DID_FLASH_CONFIG)
69  {
70  memcpy(s->flashConfig, data->buf, sizeof(nvm_flash_cfg_t));
71  }
72  }
73 }

156 void InertialSense::LoggerThread(void* info)
157 {
158  bool running = true;
159  InertialSense* inertialSense = (InertialSense*)info;
160 
161  // gather up packets in memory
162  vector<p_data_t> packets;
163 
164  while (running)
165  {
166  SLEEP_MS(2);
167  {
168  // lock so we can read and clear m_logPackets
169  cMutexLocker logMutexLocker(&inertialSense->m_logMutex);
170  for (list<p_data_t>::iterator i = inertialSense->m_logPackets.begin(); i != inertialSense->m_logPackets.end(); i++)
171  {
172  packets.push_back(*i);
173  }
174 
175  // clear shared memory
176  inertialSense->m_logPackets.clear();
177 
178  // update running state
179  running = inertialSense->m_logger.Enabled();
180  }
181 
182  // log the packets
183  for (size_t i = 0; i < packets.size(); i++)
184  {
185  if (!inertialSense->m_logger.LogData(0, &packets[i].hdr, packets[i].buf))
186  {
187  // Failed to write to log
188  return;
189  }
190  }
191  packets.clear();
192  }
193 }

206 bool InertialSense::Open(const char* port, int baudRate)
207 {
208  // clear the device in preparation for auto-baud
209  memset(&m_deviceInfo, 0, sizeof(m_deviceInfo));
210  memset(&m_config, 0, sizeof(m_config));
211  memset(&m_flashConfig, 0, sizeof(m_flashConfig));
212 
213  if (validateBaudRate(baudRate) != 0 || serialPortOpen(&m_serialPort, port, baudRate, 0) != 1)
214  {
215  return false;
216  }
217 
218  // negotiate baud rate by querying device info - don't return out until it negotiates or times out
219  // if the baud rate is already correct, the request for the message should succeed very quickly
220  time_t startTime = time(0);
221 
222  // try to auto-baud for up to 10 seconds, then abort if we didn't get a valid packet
223  // we wait until we get a valid serial number and manufacturer
224  while ((m_flashConfig.size == 0 || m_deviceInfo.serialNumber == 0 || m_deviceInfo.manufacturer[0] == '\0') && time(0) - startTime < 10)
225  {
226  getDataComManagerInstance(m_cmHandle, 0, DID_CONFIG, 0, 0, 0);
227  getDataComManagerInstance(m_cmHandle, 0, DID_DEV_INFO, 0, 0, 0);
228  getDataComManagerInstance(m_cmHandle, 0, DID_FLASH_CONFIG, 0, 0, 0);
229  SLEEP_MS(12);
230  stepComManagerInstance(m_cmHandle);
231  SLEEP_MS(12);
232  }
233 
234  if (m_flashConfig.size == 0 || m_deviceInfo.serialNumber == 0 || m_deviceInfo.manufacturer[0] == '\0')
235  {
236  Close();
237  return false;
238  }
239 
240  return true;
241 }

323 bool InertialSense::CreateHost(const string& ipAndPort)
324 {
325  // if no serial connection, fail
326  if (!IsOpen())
327  {
328  return false;
329  }
330 
331  CloseServerConnection();
332  size_t colon = ipAndPort.find(':', 0);
333  if (colon == string::npos)
334  {
335  return false;
336  }
337  StopBroadcasts();
338  string host = ipAndPort.substr(0, colon);
339  string portString = ipAndPort.substr(colon + 1);
340  int port = (int)strtol(portString.c_str(), NULL, 10);
341  bool opened = (m_tcpServer.Open(host, port) == 0);
342  if (opened)
343  {
344  // configure as RTK base station
345  uint32_t cfgBits = SYS_CFG_BITS_RTK_BASE_STATION;
346  sendDataComManagerInstance(m_cmHandle, m_pHandle, DID_FLASH_CONFIG, &cfgBits, sizeof(cfgBits), OFFSETOF(nvm_flash_cfg_t, sysCfgBits));
347  cfgBits = 1; // raw messages
348  sendDataComManagerInstance(m_cmHandle, m_pHandle, DID_CONFIG, &cfgBits, sizeof(cfgBits), OFFSETOF(config_t, msgCfgBits));
349  }
350  return opened;
351 }

358 void InertialSense::Update()
359 {
360  uint8_t buffer[4096];
361  if (m_tcpClient.IsOpen())
362  {
363  int count = m_tcpClient.Read(buffer, sizeof(buffer));
364  if (count > 0)
365  {
366  m_tcpByteCount += count;
367  m_parser->Write(buffer, count);
368  }
369  }
370  else if (m_tcpServer.IsOpen())
371  {
372  int count = serialPortRead(&m_serialPort, buffer, sizeof(buffer));
373  if (count > 0)
374  {
375  m_tcpServer.Write(buffer, count);
376  m_tcpByteCount += count;
377  }
378  m_tcpServer.Update();
379  }
380  stepComManager();
381 }

597 py::dict py_dataHandling(p_data_t* data)
598 {
599  // uDatasets is a union of all datasets that we can receive. See data_sets.h for a full list of all available datasets.
600  uDatasets d = {};
601  copyDataPToStructP(&d, data, sizeof(uDatasets));
602 
603  // Put the data into a python usable type (This works)
604  py::dict dOut;
605  py::dict header;
606  py::dict msgData;
607  header[py::str("id")] = data->hdr.id;
608  header[py::str("size")] = data->hdr.size;
609  header[py::str("offset")] = data->hdr.offset;
610  dOut[py::str("header")] = header;
611 
612  // Example of how to access dataset fields.
613  switch (data->hdr.id)
614  {
615  case DID_INS_2:
616  msgData["qn2b"] = convertArray(d.ins2.qn2b, 4); // quaternion attitude
617  msgData["uvw"] = convertArray(d.ins2.uvw, 3); // body velocities
618  msgData["lla"] = convertArray(d.ins2.lla, 3); // latitude, longitude, altitude
619  break;
620  case DID_INS_1:
621  msgData["theta"] = convertArray(d.ins1.theta, 3); // euler attitude
622  msgData["lla"] = convertArray(d.ins1.lla, 3); // latitude, longitude, altitude
623  break;
624  case DID_DUAL_IMU:
625  msgData["imu1acc"] = convertArray(d.dualImu.I[0].acc, 3);
626  msgData["imu1pqr"] = convertArray(d.dualImu.I[0].pqr, 3);
627  msgData["imu2acc"] = convertArray(d.dualImu.I[1].acc, 3);
628  msgData["imu2pqr"] = convertArray(d.dualImu.I[1].pqr, 3);
629  msgData["time"] = d.dualImu.time;
630  break;
631  case DID_DELTA_THETA_VEL:
632  msgData["time"] = d.dThetaVel.time;
633  msgData["theta"] = convertArray(d.dThetaVel.theta, 3);
634  msgData["uvw"] = convertArray(d.dThetaVel.uvw, 3);
635  msgData["dt"] = d.dThetaVel.dt;
636  break;
637  case DID_GPS:
638  {
639  msgData["rxps"] = d.gps.rxps;
640  msgData["towOffset"] = d.gps.towOffset;
641

780  {
781  vector<string> keyAndValue = splitString(keyValues[i], '=');
782  if (keyAndValue.size() == 2)

784  if (flashMap.find(keyAndValue[0]) == flashMap.end())
785  {
786  cout << "Unrecognized flash config key '" << keyAndValue[0] << "' specified, ignoring." << endl;
787  }
788  else

791  cISDataMappings::StringToData(keyAndValue[1].c_str(), NULL, (uint8_t*)&flashConfig, info);

793  }
794  }
795  }
796  inertialSenseInterface.SetFlashConfig(flashConfig);
