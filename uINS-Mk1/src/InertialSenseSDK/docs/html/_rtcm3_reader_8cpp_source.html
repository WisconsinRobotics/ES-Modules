124 void cRtcm3Reader::WriteByte(uint8_t b)
125 {
126  // if we have no bytes yet, check for header
127  // if we have exactly 3 bytes, we have enough to calculate the length
128  // fewer than 3 bytes and we want to exit out early
129  if (m_rtcm.len <= 3)
130  {
131  // check for header byte
132  if (m_rtcm.len == 0)
133  {
134  // 0xD3 is the first byte in an RTCM3 message
135  if (b != 0xD3)
136  {
137  // corrupt data
138  return;
139  }
140  }
141  else if (m_rtcm.len == 3)
142  {
143  uint32_t msgLength = getBitsAsUInt32(m_rtcm.buff, 14, 10);
144  if (msgLength > 1023)
145  {
146  // corrupt data
147  m_rtcm.len = 0;
148  return;
149  }
150  else
151  {
152  // message length only includes the message bytes, not the header or 3 CRC bytes at the end
153  // plus 2 because we will be appending the first byte of the message or CRC bytes
154  m_bytesRemaining = msgLength + 2;
155  }
156  }
157 
158  // append the byte
159  m_rtcm.buff[m_rtcm.len++] = b;
160  return;
161  }
162 
163  // append the byte
164  m_rtcm.buff[m_rtcm.len++] = b;
165 
166  // see if we are done reading bytes, if so calculate CRC and forward the message on to the uINS if it is valid
167  if (--m_bytesRemaining == 0)
168  {
169  // if the message is large enough, forward it on
170  if (m_rtcm.len > 5)
171  {
172  // remove the CRC bytes
173  m_rtcm.len -= 3;
174 
175  // the CRC calculation does not include the 3 CRC bytes at the end for obvious reasons
176  uint32_t actualCRC = calculate24BitCRCQ(m_rtcm.buff, m_rtcm.len);
177 
178  // get the CRC from the last 3 bytes of the message
179  uint32_t correctCRC = getBitsAsUInt32(m_rtcm.buff + m_rtcm.len, 0, 24);
180 
181  // if valid CRC, forward mesasge to uINS
182  if (actualCRC == correctCRC)
183  {
184  if (m_delegate->OnPacketReceived(this, m_rtcm.buff, m_rtcm.len))
185  {
186  ParseMessage();
187  }
188  }
189  else
190  {
191  // corrupt data
192  }
193  }
