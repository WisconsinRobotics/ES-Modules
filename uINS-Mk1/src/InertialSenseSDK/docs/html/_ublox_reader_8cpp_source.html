57  case 0:

63  case 1:

75  return;
76  case 3:
77  m_messageId = b;
78  case 4: // payload 1
79  case 5: // payload 2
80  m_raw.buff[m_raw.len++] = b;
81  return;
82  case UBLOX_HEADER_SIZE:
83  // add 2 bytes for the checksum
84  m_raw.nbyte = BE_SWAP16(*(uint16_t*)(m_raw.buff + 4)) + UBLOX_CHECKSUM_SIZE;
85 
86  // if packet size is larger than buffer minus header and checksum
87  if (m_raw.nbyte < UBLOX_CHECKSUM_SIZE || (unsigned int)m_raw.nbyte > sizeof(m_raw.buff) - UBLOX_HEADER_SIZE)
88  {
89  // corrupt data
90  Reset();
91  return;
92  }
93  break;
94  }
95  m_raw.buff[m_raw.len++] = b;
96  if (--m_raw.nbyte == 0)
97  {
98  // get actual checksum
99  uint8_t checksum1 = m_raw.buff[m_raw.len - 2];
100  uint8_t checksum2 = m_raw.buff[m_raw.len - 1];
101  uint8_t actualChecksum1, actualChecksum2;
102  // checksum does not include the two preamble bytes
103  CalculateChecksum(m_raw.buff, m_raw.len, actualChecksum1, actualChecksum2);
104  if (checksum1 == actualChecksum1 && checksum2 == actualChecksum2)
105  {
106  if (m_delegate->OnPacketReceived(this, m_raw.buff, m_raw.len))
107  {
108  ParseMessage();
109  }
110  }
111  else

114  }

116  }
