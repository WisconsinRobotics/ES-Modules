250 bool cltool_replayDataLog()
251 {
252  if (g_commandLineOptions.logPath.length() == 0)
253  {
254  cout << "Please specify the replay log path!" << endl;
255  return false;
256  }
257 
258  cISLogger logger;
259  if (!logger.LoadFromDirectory(g_commandLineOptions.logPath))
260  {
261  cout << "Failed to load log files: " << g_commandLineOptions.logPath << endl;
262  return false;
263  }
264 
265  cout << "Replaying log files: " << g_commandLineOptions.logPath << endl;
266  p_data_t *data;
267  while ((data = logger.ReadData()) != NULL)
268  {
269  g_inertialSenseDisplay.ProcessData(data, g_commandLineOptions.replayDataLog, g_commandLineOptions.replaySpeed);
270  }
271 
272  cout << "Done replaying log files: " << g_commandLineOptions.logPath << endl;
273  g_inertialSenseDisplay.Goodbye();
274  return true;
275 }

346 bool cltool_updateFlashConfig(InertialSense& inertialSenseInterface, string flashConfigString)
347 {
348  const nvm_flash_cfg_t& flashConfig = inertialSenseInterface.GetFlashConfig();
349  const map_lookup_name_t& globalMap = cISDataMappings::GetMap();
350  const map_name_to_info_t& flashMap = globalMap.at(DID_FLASH_CONFIG);
351 
352  if (flashConfigString.length() < 2)
353  {
354  // read flash config and display
355  data_mapping_string_t stringBuffer;
356  cout << "Current flash config" << endl;
357  for (map_name_to_info_t::const_iterator i = flashMap.begin(); i != flashMap.end(); i++)
358  {
359  if (cISDataMappings::DataToString(i->second, NULL, (const uint8_t*)&flashConfig, stringBuffer))
360  {
361  cout << i->second.name << " = " << stringBuffer << endl;
362  }
363  }
364  return false;
365  }
366  else
367  {
368  nvm_flash_cfg_t flashConfig = inertialSenseInterface.GetFlashConfig();
369  vector<string> keyValues = splitString(flashConfigString, '|');
370  for (size_t i = 0; i < keyValues.size(); i++)
371  {
372  vector<string> keyAndValue = splitString(keyValues[i], '=');
373  if (keyAndValue.size() == 2)
374  {
375  if (flashMap.find(keyAndValue[0]) == flashMap.end())
376  {
377  cout << "Unrecognized flash config key '" << keyAndValue[0] << "' specified, ignoring." << endl;
378  }
379  else
380  {
381  const data_info_t& info = flashMap.at(keyAndValue[0]);
382  cISDataMappings::StringToData(keyAndValue[1].c_str(), keyAndValue[1].length(), NULL, (uint8_t*)&flashConfig, info);
383  cout << "Updated flash config key '" << keyAndValue[0] << "' to '" << keyAndValue[1].c_str() << "'" << endl;
384  }
385  }
386  }
387  inertialSenseInterface.SetFlashConfig(flashConfig);
388  SLEEP_MS(1000);
389  g_inertialSenseDisplay.Clear();
390  return true;
391  }
