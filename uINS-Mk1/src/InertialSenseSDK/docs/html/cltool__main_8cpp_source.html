47 static bool cltool_setupCommunications(InertialSense& inertialSenseInterface)
48 {
49  int periodMs = 50;
50  inertialSenseInterface.StopBroadcasts(); // Stop streaming any prior messages
51 
52  // ask for device info every 2 seconds
53  inertialSenseInterface.BroadcastBinaryData(DID_DEV_INFO, 2000, cltool_dataCallback);
54 
55  // depending on command line options. stream various data sets
56  if (g_commandLineOptions.streamINS1)
57  {
58  inertialSenseInterface.BroadcastBinaryData(DID_INS_1, periodMs, cltool_dataCallback);
59  }
60  if (g_commandLineOptions.streamINS2)
61  {
62  inertialSenseInterface.BroadcastBinaryData(DID_INS_2, periodMs, cltool_dataCallback);
63  }
64  if (g_commandLineOptions.streamSysSensors)
65  {
66  inertialSenseInterface.BroadcastBinaryData(DID_SYS_SENSORS, 100, cltool_dataCallback);
67  }
68  if (g_commandLineOptions.streamDualIMU)
69  {
70  inertialSenseInterface.BroadcastBinaryData(DID_DUAL_IMU, periodMs, cltool_dataCallback);
71  }
72  if (g_commandLineOptions.streamIMU1)
73  {
74  inertialSenseInterface.BroadcastBinaryData(DID_IMU_1, periodMs, cltool_dataCallback);
75  }
76  if (g_commandLineOptions.streamIMU2)
77  {
78  inertialSenseInterface.BroadcastBinaryData(DID_IMU_2, periodMs, cltool_dataCallback);
79  }
80  if (g_commandLineOptions.streamDThetaVel)
81  {
82  inertialSenseInterface.BroadcastBinaryData(DID_DELTA_THETA_VEL, periodMs, cltool_dataCallback);
83  }
84  if (g_commandLineOptions.streamGPS)
85  {
86  inertialSenseInterface.BroadcastBinaryData(DID_GPS, 200, cltool_dataCallback);
87  }
88  if (g_commandLineOptions.streamMag1)
89  {
90  inertialSenseInterface.BroadcastBinaryData(DID_MAGNETOMETER_1, periodMs, cltool_dataCallback);
91  }
92  if (g_commandLineOptions.streamBaro)
93  {
94  inertialSenseInterface.BroadcastBinaryData(DID_BAROMETER, periodMs, cltool_dataCallback);
95  }
96  if (g_commandLineOptions.logSolution != 0)
97  {
98  config_t cfg = inertialSenseInterface.GetConfig();
99  cfg.sLogCtrl = g_commandLineOptions.logSolution;
100  inertialSenseInterface.SetConfig(cfg);
101  }
102  if (g_commandLineOptions.serverConnection.length() != 0)
103  {
104  if (g_commandLineOptions.serverConnection.find("RTCM3:") == 0)
105  {
106  if (!inertialSenseInterface.OpenServerConnectionRTCM3(g_commandLineOptions.serverConnection.substr(6)))
107  {
108  cout << "Failed to connect to server." << endl;
109  }
110  }
111  else if (g_commandLineOptions.serverConnection.find("IS:") == 0)
112  {
113  if (!inertialSenseInterface.OpenServerConnectionInertialSense(g_commandLineOptions.serverConnection.substr(3)))
114  {
115  cout << "Failed to connect to server." << endl;
116  }
117  }
118  else if (g_commandLineOptions.serverConnection.find("UBLOX:") == 0)
119  {
120  if (!inertialSenseInterface.OpenServerConnectionUblox(g_commandLineOptions.serverConnection.substr(6)))
121  {
122  cout << "Failed to connect to server." << endl;
123  }
124  }
125  else
126  {
127  cout << "Invalid server connection, must prefix with RTCM3: or IS:, " << g_commandLineOptions.serverConnection << endl;
128  return false;
129  }
130  }
131  if (g_commandLineOptions.flashConfig.length() != 0)
132  {
133  return cltool_updateFlashConfig(inertialSenseInterface, g_commandLineOptions.flashConfig);
134  }
135  return true;
136 }

152 static int cltool_createHost()
153 {
154  InertialSense inertialSenseInterface;
155  if (!inertialSenseInterface.Open(g_commandLineOptions.comPort.c_str(), g_commandLineOptions.baudRate))
156  {
157  cout << "Failed to open serial port at " << g_commandLineOptions.comPort.c_str() << endl;
158  return -1; // Failed to open serial port
159  }
160  else if (g_commandLineOptions.flashConfig.length() != 0 && !cltool_updateFlashConfig(inertialSenseInterface, g_commandLineOptions.flashConfig))
161  {
162  return -1;
163  }
164  else if (!inertialSenseInterface.CreateHost(g_commandLineOptions.host))
165  {
166  cout << "Failed to create host at " << g_commandLineOptions.host << endl;
167  return -1; // Failed to open host
168  }
169  while (!g_inertialSenseDisplay.ControlCWasPressed())
170  {
171  inertialSenseInterface.Update();
172  g_inertialSenseDisplay.Home();
173  cout << "Tcp bytes sent: " << inertialSenseInterface.GetTcpByteCount() << " ";
174  SLEEP_MS(1);
175  }
176  cout << "Shutting down..." << endl;
177 
178  // close the interface cleanly, this ensures serial port and any logging are shutdown properly
179  inertialSenseInterface.Close();
180 
181  return 0;
182 }

184 static int inertialSenseMain()
185 {
186  // clear display
187  g_inertialSenseDisplay.SetDisplayMode(g_commandLineOptions.displayMode);
188  g_inertialSenseDisplay.Clear();
189 
190  // if replay data log specified on command line, do that now and return
191  if (g_commandLineOptions.replayDataLog)
192  {
193  // [REPLAY INSTRUCTION] 1.) Replay data log
194  return !cltool_replayDataLog();
195  }
196 
197  // if bootloader was specified on the command line, do that now and return out
198  else if (g_commandLineOptions.bootloaderFileName.length() != 0)
199  {
200  // [BOOTLOADER INSTRUCTION] 1.) Run bootloader
201  return cltool_runBootloader();
202  }
203 
204  // if host was specified on the command line, create a tcp server
205  else if (g_commandLineOptions.host.length() != 0)
206  {
207  return cltool_createHost();
208  }
209 
210  // open the device, start streaming data and logging
211  else
212  {
213  // [COMM INSTRUCTION] 1.) Create InertialSense object and open serial port
214  InertialSense inertialSenseInterface(cltool_dataCallback);
215  if (!inertialSenseInterface.Open(g_commandLineOptions.comPort.c_str(), g_commandLineOptions.baudRate))
216  {
217  cout << "Failed to open serial port at " << g_commandLineOptions.comPort.c_str() << endl;
218  return -1; // Failed to open serial port
219  }
220 
221  // [COMM INSTRUCTION] 2.) Enable data broadcasting from uINS
222  if (cltool_setupCommunications(inertialSenseInterface))
223  {
224  // [LOGGER INSTRUCTION] Setup and start data logger
225  if (!cltool_setupLogger(inertialSenseInterface))
226  {
227  cout << "Failed to setup logger!" << endl;
228  inertialSenseInterface.Close();
229  return -1;
230  }
231  try
232  {
233  // Main loop. Could be in separate thread if desired.
234  while (!g_inertialSenseDisplay.ControlCWasPressed())
235  {
236  // [COMM INSTRUCTION] 3.) Process data and messages
237  inertialSenseInterface.Update();
238 
239  if (inertialSenseInterface.GetTcpByteCount() != 0)
240  {
241  g_inertialSenseDisplay.GoToRow(1);
242  cout << "Tcp bytes read: " << inertialSenseInterface.GetTcpByteCount() << " ";
243  }
244 
245  // Specify the minimum time between read/write updates.
246  SLEEP_MS(1);
247  }
248  }
249  catch (...)
250  {
251  cout << "Unknown exception, ";
252  }
253  }
254 
255  cout << "Shutting down..." << endl;
256 
257  // close the interface cleanly, this ensures serial port and any logging are shutdown properly
258  inertialSenseInterface.Close();
259  }
260 
261  return 0;
262 }
